    <style>
        #contador-descendente {
            font-size: 2em;
            font-weight: bold;
            color: red;
        }
    </style>
    <script>
        let contador = 0;
        let contadorDescendente = 10;
        let intervalo;

        function iniciarContadorDescendente() {
            if (intervalo) {
                clearInterval(intervalo);
            }
            contadorDescendente = 10;
            document.getElementById("contador-descendente").innerText = "Compte enrrera: " + contadorDescendente;

            intervalo = setInterval(function() {
                if (contadorDescendente > 0) {
                    contadorDescendente--;
                    document.getElementById("contador-descendente").innerText = "Compte enrrera: " + contadorDescendente;
                } else {
                    clearInterval(intervalo);
                    document.getElementById("contador-descendente").innerText = "¡TEMPS ACABATS!";
                }
            }, 1000); // Actualiza cada segundo
        }

        function incrementarContador() {
            contador++;
            document.getElementById("contador").innerText = "Pardalillos que apreten: " + contador;
            document.getElementById("mensaje").innerText = "¡Has apretat el botó!";
            document.getElementById("advertencia").innerText = "Atenció, es comenca a borrar la memoria del equip, et queden 10 segons.";
            iniciarContadorDescendente();
        }
    </script>
</head>

    <p id="mensaje">Apreta el botó:</p>
    <button onclick="incrementarContador()">Apretem</button>

    <p id="advertencia"></p> <!-- Elemento para el mensaje de advertencia -->
